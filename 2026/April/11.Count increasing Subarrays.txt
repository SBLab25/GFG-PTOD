class Solution {
    public int countIncreasing(int[] arr) {
        // code here
        int count=1;
        int n=arr.length;
        int ans=0;
        int j=n;
        for(int i=n-1;i>0;i--)
        {
            if(arr[i-1]<arr[i])
            {
              
               ans+=j-i;
            }else{
               
                j=i;
            }
        }
        
        return ans;
    }
}
