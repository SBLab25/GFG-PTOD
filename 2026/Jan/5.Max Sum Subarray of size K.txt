class Solution {
    public int maxSubarraySum(int[] arr, int k) {
        // Code here
        int answer=0;
        int n=arr.length;
        if(n<k)
        return 0;
        
        int sum=0;
        for(int i=0;i<k;i++)
        {
           sum+=arr[i]; 
        }
        answer=sum;
        for(int i=k;i<n;i++)
        {
            sum=sum+(arr[i]-arr[i-k]);
            answer=Math.max(sum,answer);
        }
        return answer;
    }
}
